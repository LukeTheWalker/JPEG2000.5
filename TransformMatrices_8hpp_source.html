23 constexpr auto generate_forward(const std::array<T, N>& arr) {
24 std::array<T, N * 2> forward;
25 for (std::size_t i = 0; i < N; ++i)
26 forward[i] = arr[i] / 2;
27 for (std::size_t i = 0; i < N; ++i)
28 forward[N + i] = (i % 2 == 1 ? -1 : 1) * arr[N - i - 1] / 2;
29 return forward;
30 }

39 template <typename T, std::size_t N>
40 constexpr auto generate_inverse(const std::array<T, N>& arr) {
41 std::array<T, N> inverse;
42 for (std::size_t i = N / 2 - 2; (int)i >= 0; i -= 2) {
43 inverse[N / 2 - i - 2] = arr[i] * 2;
44 inverse[N / 2 - i - 1] = arr[i + N / 2] * 2;
45 }
46 for (std::size_t i = N / 2 - 1; (int)i >= 1; i -= 2) {
47 inverse[N - i - 1] = arr[i] * 2;
48 inverse[N - i] = arr[i + N / 2] * 2;
49 }
50 return inverse;
51 }
