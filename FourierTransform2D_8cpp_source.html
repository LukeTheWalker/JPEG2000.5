108void FourierTransform2D<FT>::OutputSpace::pass_filter(const double cutoff_perc, const bool is_high_pass)
109{
110 auto channels = data.size();
111 auto rows = data[0].size();
112 auto cols = data[0][0].size();
113
114 unsigned long int area_square = rows * cols;
115 // r = √((g/100) × x^2 / π)
116 auto cutoff_freq = sqrt((1 - cutoff_perc) * (double)area_square / M_PI);
117
118 pair<int, int> center = {rows / 2, cols / 2};
119 // lamda for distance calculation
120 auto distance = [&](int i, int j) { return sqrt(pow(i - center.first, 2) + pow(j - center.second, 2)); };
121
122 // Apply the low-pass filter
123 for (auto c = 0ull; c < channels; ++c)
124 for (auto i = 0ull; i < rows; ++i)
125 for (auto j = 0ull; j < cols; ++j)
126 if ((is_high_pass && distance(i,j) > cutoff_freq) || (!is_high_pass && distance(i,j) <= cutoff_freq))
127 data[c][i][j] = 0.0;
128}

131void FourierTransform2D<FT>::OutputSpace::magnitude_filter(double cutoff_percentage) {
132 auto channels = data.size();
133 auto rows = data[0].size();
134 auto cols = data[0][0].size();
135
136 // Calculate the magnitude of the FFT
137 vector<double> magnitude(rows * cols * channels);
138 for (auto c = 0ull; c < channels; ++c)
139 for (auto i = 0ull; i < rows; ++i)
140 for (auto j = 0ull; j < cols; ++j)
141 magnitude[i * cols + j + c * rows * cols] = abs(data[c][i][j]);
142
143 // Sort the magnitude vector
144 sort(magnitude.begin(), magnitude.end());
145
146 // Calculate the cutoff value
147 double cutoff_value = magnitude[(int)(magnitude.size() * cutoff_percentage)];
148
149 // Apply the filter
150 for (auto c = 0ull; c < channels; ++c)
151 for (auto i = 0ull; i < rows; ++i)
152 for (auto j = 0ull; j < cols; ++j)
153 if (abs(data[c][i][j]) < cutoff_value)
154 data[c][i][j] = 0.0;
155}

158auto FourierTransform2D<FT>::compute2DFFT(Typedefs::vcpx3D& fft_coeff, const bool is_inverse) const -> void {
159 auto channels = fft_coeff.size();
160 auto rows = fft_coeff[0].size();
161 auto cols = fft_coeff[0][0].size();
162
163 for (auto c = 0ull; c < channels; ++c){
164
165 // Compute the 1D FFT along each row
166 for (auto i = 0ull; i < rows; ++i){
167 auto tmp_vec = fft_coeff[c][i];
168 ft(tmp_vec, is_inverse); // Compute the FFT
169 fft_coeff[c][i]= tmp_vec;
170 }
171
172 // Compute the 1D FFT along each column
173 for (auto j = 0ull; j < cols; ++j)
174 {
175 Typedefs::vcpx col(rows);
176 for (auto i = 0ull; i < rows; ++i) col[i] = fft_coeff[c][i][j];
177 ft(col, is_inverse); // Compute the FFT
178 for (auto i = 0ull; i < rows; ++i) fft_coeff[c][i][j] = col[i];
179 }
180 }
181}
